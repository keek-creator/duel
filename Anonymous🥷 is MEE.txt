local function createAnimatedDot(frame, startPosX, startPosY, speed, trans)
    local dot = Instance.new('Frame')
    dot.Size = UDim2.new(0, 2, 0, 2)
    dot.Position = UDim2.new(startPosX, 0, startPosY, 0)
    dot.BackgroundColor3 = Color3.new(1, 1, 1)
    dot.BackgroundTransparency = trans or 0.65
    dot.BorderSizePixel = 0
    dot.ZIndex = 2
    dot.Parent = frame
    
    local dotCorner = Instance.new('UICorner')
    dotCorner.CornerRadius = UDim.new(1, 0)
    dotCorner.Parent = dot
    
    table.insert(animatedDots, {
        dot = dot,
        startY = startPosY,
        speed = speed,
        phase = math.random() * 100
    })
    
    return dot
end

local lastTime = tick()
RunService.Heartbeat:Connect(function()
    local currentTime = tick()
    local dt = currentTime - lastTime
    lastTime = currentTime
    
    for _, dotData in ipairs(animatedDots) do
        if dotData.speed > 0 then
            local dot = dotData.dot
            local currentPos = dot.Position.Y.Scale
            local newY = currentPos + (dotData.speed * dt * 0.5)
            
            if newY > 1.05 then
                newY = -0.05
            end
            
            dot.Position = UDim2.new(dot.Position.X.Scale, 0, newY, 0)
        end
    end
end)
