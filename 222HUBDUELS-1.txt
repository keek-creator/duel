    local b = Instance.new("TextButton")

    b.Parent = parent or MainPage

    local corner = Instance.new("UICorner", b)
    corner.CornerRadius = UDim.new(0, 8)

    local stroke = Instance.new("UIStroke", b)
    stroke.Color = Color3.fromRGB(50, 255, 150)
    stroke.Thickness = 0
    stroke.Transparency = 0.7

    b.MouseEnter:Connect(function()
        TweenService:Create(b, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
            BackgroundColor3 = Color3.fromRGB(35, 45, 50)
        }):Play()
        TweenService:Create(stroke, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
            Thickness = 2
        }):Play()
    end)

    b.MouseLeave:Connect(function()
        TweenService:Create(b, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
            BackgroundColor3 = Color3.fromRGB(25, 30, 35)
        }):Play()
        TweenService:Create(stroke, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
            Thickness = 0
        }):Play()
    end)

    return b
end

local CONFIG_FILE = "222Hub_Config.json"
local FloatFrames = {}

local function flashSaved()
    SavedLabel.Text = "✓ Config Saved"
    task.delay(1.5, function() SavedLabel.Text = "" end)
end

}

local HttpService = game:GetService("HttpService")

local function saveConfig()
    LiveCfg.floatPositions = {}
    for name, frame in pairs(FloatFrames) do
        LiveCfg.floatPositions[name] = {
            x = frame.Position.X.Scale,
            y = frame.Position.Y.Scale,
        }
    end
    local ok, err = pcall(function()
        writefile(CONFIG_FILE, HttpService:JSONEncode(LiveCfg))
    end)
    if ok then flashSaved() else warn("[222Hub] Save failed: " .. tostring(err)) end
end

local CFG_VERSION = 2

local function loadConfig()
    local ok, data = pcall(function()
        if not isfile(CONFIG_FILE) then return nil end
        local raw = readfile(CONFIG_FILE)
        return HttpService:JSONDecode(raw)
    end)
    if not ok or not data then
        print("[222Hub] No saved config found, using defaults.")
        return nil
    end
    if data.version ~= CFG_VERSION then
        print("[222Hub] Old config version detected, resetting.")
        pcall(function() delfile(CONFIG_FILE) end)
        return nil
    end
    return data
end

local function bindKey(box, keyRefName)
    box.FocusLost:Connect(function()
        local inputName = box.Text
        if inputName and #inputName > 0 then
            local enumKey = Enum.KeyCode[inputName:upper()]
            if enumKey then
                if keyRefName == "GuiToggleKey" then GuiToggleKey = enumKey
                elseif keyRefName == "SpeedKey" then SpeedKey = enumKey
                elseif keyRefName == "SpinKey" then SpinKey = enumKey
                end
                box.PlaceholderText = enumKey.Name
                box.Text = ""
            else
                box.Text = ""
                box.PlaceholderText = "Invalid Key"
            end
        end
    end)
end

local SpeedConnection

function enableSpeedBypass()
    if SpeedConnection then SpeedConnection:Disconnect() end
    SpeedConnection = RunService.Stepped:Connect(function()
        if not SpeedEnabled then return end
        local char = LocalPlayer.Character
        if not char then return end
        local hum = char:FindFirstChildOfClass("Humanoid")
        local root = char:FindFirstChild("HumanoidRootPart")
        if not hum or not root then return end
        if hum:GetState() == Enum.HumanoidStateType.Running then
            local moveDir = hum.MoveDirection
            if moveDir.Magnitude > 0 then
                root.Velocity = Vector3.new(moveDir.X * SpeedValue, root.Velocity.Y, moveDir.Z * SpeedValue)
            end
        end
    end)
end

function disableSpeedBypass()
    if SpeedConnection then
        SpeedConnection:Disconnect()
        SpeedConnection = nil
    end
end

local ESPs = {}

local function CreateESPBox(player)
    if player == LocalPlayer then return end
    if ESPs[player] then return end
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChild("Humanoid")
    if not hrp or not hum then return end
    local box = Instance.new("BoxHandleAdornment")
    box.Adornee = hrp
    box.Size = Vector3.new(3, hum.HipHeight * 2.5, 3)
    box.Color3 = Color3.fromRGB(255, 0, 0)
    box.Transparency = 0.5
    box.AlwaysOnTop = true
    box.ZIndex = 10
    box.Parent = hrp
    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = hrp
    billboard.AlwaysOnTop = true
    billboard.Size = UDim2.new(0, 100, 0, 25)
    billboard.StudsOffset = Vector3.new(0, hum.HipHeight + 2, 0)
    billboard.Parent = hrp
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 1, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = player.Name
    nameLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
    nameLabel.Font = Enum.Font.GothamBold
    nameLabel.TextScaled = true
    nameLabel.Parent = billboard
    ESPs[player] = {Box = box, Billboard = billboard}
end

local function RemoveESPBox(player)
    local data = ESPs[player]
    if data then
        if data.Box then data.Box:Destroy() end
        if data.Billboard then data.Billboard:Destroy() end
        ESPs[player] = nil
    end
end

    ESPEnabled = not ESPEnabled
    UpdateButtonState(ESPBtn, ESPEnabled, "Player ESP [ON]", "Player ESP [OFF]")
    if not ESPEnabled then
        for player, _ in pairs(ESPs) do RemoveESPBox(player) end
    end
    LiveCfg.ESPEnabled = ESPEnabled
end

-- ================= AUTO GRAB (Orrxl4 nearest-steal logic) =================
-- Finds the nearest ProximityPrompt with ActionText == "Steal" within grabRadius,
-- sets HoldDuration = 0, RequiresLineOfSight = false, fires it every heartbeat.
do
    local grabRadius = 50
    local autoGrabEnabled = false
    local grabPrompts = {}
    local grabConnections = {}

    local function grabAddConnection(conn)
        table.insert(grabConnections, conn)
    end

    local function grabAddPrompt(obj)
        if obj:IsA("ProximityPrompt") then
            table.insert(grabPrompts, obj)
        end
    end

    local function grabRemovePrompt(obj)
        for i = #grabPrompts, 1, -1 do
            if grabPrompts[i] == obj then table.remove(grabPrompts, i) end
        end
    end

    local function findNearestSteal(rootPos)
        local nearest, dist = nil, math.huge
        for _, prompt in ipairs(grabPrompts) do
            if prompt and prompt.Parent and prompt.Enabled and prompt.ActionText == "Steal" then
                local part = prompt.Parent:IsA("BasePart") and prompt.Parent
                    or prompt:FindFirstAncestorWhichIsA("BasePart")
                if part then
                    local d = (part.Position - rootPos).Magnitude
                    if d < dist and d <= grabRadius then
                        nearest = prompt
                        dist = d
                    end
                end
            end
        end
        return nearest
    end

    -- Seed existing prompts
    coroutine.wrap(function()
        for _, target in {workspace, ReplicatedStorage} do
            for _, obj in ipairs(target:GetDescendants()) do
                grabAddPrompt(obj)
            end
        end
    end)()

    grabAddConnection(workspace.DescendantAdded:Connect(grabAddPrompt))
    grabAddConnection(workspace.DescendantRemoving:Connect(grabRemovePrompt))
    grabAddConnection(ReplicatedStorage.DescendantAdded:Connect(grabAddPrompt))
    grabAddConnection(ReplicatedStorage.DescendantRemoving:Connect(grabRemovePrompt))

    -- Heartbeat: find nearest steal prompt in range and fire immediately
    grabAddConnection(RunService.Heartbeat:Connect(function()
        if not autoGrabEnabled then return end
        local char = LocalPlayer.Character
        if not char then return end
        local root = char:FindFirstChild("HumanoidRootPart")
        if not root then return end

        -- Clean dead prompts
        for i = #grabPrompts, 1, -1 do
            if not grabPrompts[i] or not grabPrompts[i].Parent then
                table.remove(grabPrompts, i)
            end
        end

        local prompt = findNearestSteal(root.Position)
        if prompt then
            prompt.HoldDuration = 0
            prompt.RequiresLineOfSight = false
            prompt.Enabled = true
            pcall(function() fireproximityprompt(prompt) end)
        end
    end))

    AutoGrabBtn.MouseButton1Click:Connect(function()
        autoGrabEnabled = not autoGrabEnabled
        UpdateButtonState(AutoGrabBtn, autoGrabEnabled, "Insta Grab [ON]", "Insta Grab [OFF]")
        LiveCfg.AutoSteal = autoGrabEnabled
    end)

    -- Expose setter for config load (needed because autoGrabEnabled is upvalue inside do-block)
    _G._222Hub_autoGrabEnabled_set = function(v)
        autoGrabEnabled = v
        UpdateButtonState(AutoGrabBtn, autoGrabEnabled, "Insta Grab [ON]", "Insta Grab [OFF]")
        LiveCfg.AutoSteal = autoGrabEnabled
    end
end -- Auto Grab

-- ================= AUTO DUEL =================
-- Changes applied:
--   1. No isHoldingBrainrot gating — always go to enemy podium, return home immediately after arrival.
--   2. Interrupted path (success=false) sets isDueling=false and returningHome=false so the
--      heartbeat loop retries the enemy podium on the very next tick.
--   3. Unstuck detection: if position barely moves for 2s, nudge sideways and recompute path.
--   4. Tighter pathfinding params (AgentRadius=2, WaypointSpacing=4) to reduce wall clipping.
local PODIUM_10_WALK_SPEED = 58
local RETURN_SPEED = 29
local isDueling = false
local returningHome = false
local currentDuelTarget = nil
local activePathConnection = nil

local function stopAllPathfinding()

        activePathConnection:Disconnect()
        activePathConnection = nil
    end
    local char = LocalPlayer.Character
    if char then
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then hrp.AssemblyLinearVelocity = Vector3.zero end
    end
end

local function resetAutoDuelState()
    stopAllPathfinding()
    isDueling = false
    returningHome = false
    currentDuelTarget = nil
end

do
local function getMyPlotCashPad()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    for _, plot in ipairs(plots:GetChildren()) do
        local sign = plot:FindFirstChild("PlotSign")
        if sign then
            local yourBase = sign:FindFirstChild("YourBase")
            if yourBase and yourBase:IsA("BillboardGui") and yourBase.Enabled == true then
                local cashPad = plot:FindFirstChild("CashPad")
                if cashPad and cashPad:IsA("BasePart") then
                    return cashPad.Position
                elseif cashPad and cashPad:IsA("Model") then
                    local primary = cashPad.PrimaryPart or cashPad:FindFirstChildWhichIsA("BasePart")
                    if primary then return primary.Position end
                end
                for _, descendant in ipairs(plot:GetDescendants()) do
                    if descendant:IsA("BasePart") and (
                        descendant.Name:lower():find("cash") or
                        descendant.Name:lower():find("money") or
                        descendant.Name:lower():find("register") or
                        descendant.Name:lower():find("deposit")) then
                        return descendant.Position
                    end
                end
                local plotPrimary = plot.PrimaryPart or plot:FindFirstChildWhichIsA("BasePart")
                if plotPrimary then return plotPrimary.Position end
            end
        end
    end
    return nil
end

local function tweenToPosition(targetPos, customSpeed, onComplete)
    if typeof(targetPos) ~= "Vector3" then
        if onComplete then onComplete(false) end
        return false
    end
    local char = LocalPlayer.Character
    if not char then
        if onComplete then onComplete(false) end
        return false
    end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum then
        if onComplete then onComplete(false) end
        return false
    end
    local moveSpeed = customSpeed or PODIUM_10_WALK_SPEED
    if math.abs(hrp.Position.Y - targetPos.Y) > 50 then
        if onComplete then onComplete(false) end
        return false
    end

    local PathfindingService = game:GetService("PathfindingService")

    -- Inner helper: compute a fresh path and return waypoints, or nil on failure
    local function computePath(fromPos)
        local p = PathfindingService:CreatePath({
            AgentRadius = 2, AgentHeight = 6,
            AgentCanJump = true, AgentCanClimb = false, WaypointSpacing = 4
        })
        local ok = pcall(function() p:ComputeAsync(fromPos, targetPos) end)
        if ok and p.Status == Enum.PathStatus.Success then
            return p:GetWaypoints()
        end
        return nil
    end

    local waypoints = computePath(hrp.Position)
    if not waypoints then
        if onComplete then onComplete(false) end
        return false
    end

    local currentWaypointIndex = 1
    local startTime = tick()

    -- Unstuck tracking
    local lastStuckCheckPos  = hrp.Position
    local lastStuckCheckTime = tick()
    local nudgeCount = 0

    if activePathConnection then
        activePathConnection:Disconnect()
        activePathConnection = nil
    end

    local connection
    connection = RunService.Heartbeat:Connect(function()
        -- Stop entirely if duel was disabled
        if not AutoDuelEnabled then
            connection:Disconnect()
            if activePathConnection == connection then activePathConnection = nil end
            local currentHrp = char:FindFirstChild("HumanoidRootPart")
            if currentHrp then currentHrp.AssemblyLinearVelocity = Vector3.zero end
            if onComplete then onComplete(false) end
            return
        end
        if not char or not char.Parent then
            connection:Disconnect()
            if activePathConnection == connection then activePathConnection = nil end
            if onComplete then onComplete(false) end
            return
        end
        local currentHrp = char:FindFirstChild("HumanoidRootPart")
        local currentHum = char:FindFirstChildOfClass("Humanoid")
        if not currentHrp or not currentHum then
            connection:Disconnect()
            if activePathConnection == connection then activePathConnection = nil end
            if onComplete then onComplete(false) end
            return
        end

        -- Keep health up while travelling
        if AutoDuelEnabled and currentHum.Health <= 0 then currentHum.Health = currentHum.MaxHealth end

        -- Clamp extreme downward velocity
        if currentHrp.AssemblyLinearVelocity.Y < -80 then
            currentHrp.AssemblyLinearVelocity = Vector3.new(
                currentHrp.AssemblyLinearVelocity.X * 0.5, -40,
                currentHrp.AssemblyLinearVelocity.Z * 0.5)
        end

        -- Global timeout
        if tick() - startTime > 60 then
            connection:Disconnect()
            if activePathConnection == connection then activePathConnection = nil end
            resetAutoDuelState()
            if onComplete then onComplete(false) end
            return
        end

        -- ── UNSTUCK: every 2 s check if we barely moved ──
        local now = tick()
        if now - lastStuckCheckTime >= 2 then
            local moved = (currentHrp.Position - lastStuckCheckPos).Magnitude
            if moved < 2 and currentWaypointIndex <= #waypoints then
                nudgeCount = nudgeCount + 1
                -- Alternate nudge direction each attempt
                local nudge = (nudgeCount % 2 == 0)
                    and Vector3.new(3, 0, 0)
                    or  Vector3.new(-3, 0, 0)
                currentHrp.CFrame = currentHrp.CFrame + nudge
                -- Recompute path from nudged position
                local newWaypoints = computePath(currentHrp.Position)
                if newWaypoints then
                    waypoints = newWaypoints
                    currentWaypointIndex = 1
                end
            end
            lastStuckCheckPos  = currentHrp.Position
            lastStuckCheckTime = now
        end

        -- Finished all waypoints → success
        if currentWaypointIndex > #waypoints then
            currentHrp.AssemblyLinearVelocity = Vector3.zero
            connection:Disconnect()
            if activePathConnection == connection then activePathConnection = nil end
            if onComplete then onComplete(true) end
            return
        end

        local targetWaypoint = waypoints[currentWaypointIndex]
        if targetWaypoint.Action == Enum.PathWaypointAction.Jump then
            local heightDiff = targetWaypoint.Position.Y - currentHrp.Position.Y
            if heightDiff > 0 and heightDiff < 15 then
                currentHum:ChangeState(Enum.HumanoidStateType.Jumping)
            end
        end

        local horizontalDist = (
            Vector3.new(currentHrp.Position.X, 0, currentHrp.Position.Z) -
            Vector3.new(targetWaypoint.Position.X, 0, targetWaypoint.Position.Z)
        ).Magnitude
        if horizontalDist < 3 then
            currentWaypointIndex = currentWaypointIndex + 1
            return
        end

        local direction = (targetWaypoint.Position - currentHrp.Position) * Vector3.new(1, 0, 1)
        if direction.Magnitude > 0 then
            direction = direction.Unit
            currentHrp.AssemblyLinearVelocity = Vector3.new(
                direction.X * moveSpeed,
                currentHrp.AssemblyLinearVelocity.Y,
                direction.Z * moveSpeed)
        end
    end)

    activePathConnection = connection
    return true
end

local function getNearestEnemyPlot()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    local nearest = nil
    local minDist = math.huge
    for _, plot in ipairs(plots:GetChildren()) do
        if not plot:IsA("Model") then continue end
        local sign = plot:FindFirstChild("PlotSign")
        if sign then
            local yourBase = sign:FindFirstChild("YourBase")
            if yourBase and yourBase:IsA("BillboardGui") and yourBase.Enabled == true then continue end
        end
        local podiums = plot:FindFirstChild("AnimalPodiums")
        if podiums then
            local podium1  = podiums:FindFirstChild("1")

            if podium1 and podium10 then
                local dist1  = (hrp.Position - podium1:GetPivot().Position).Magnitude
                local dist10 = (hrp.Position - podium10:GetPivot().Position).Magnitude
                local targetPodium, targetDist
                if dist1 < dist10 then
                    targetPodium = podium1; targetDist = dist1
                else
                    targetPodium = podium10; targetDist = dist10
                end
                if targetDist < minDist then
                    minDist = targetDist
                    nearest = {plot = plot, podium = targetPodium}
                end
            end
        end
    end
    return nearest
end

-- ── Main duel heartbeat loop ──
-- Phase A (returningHome=false): walk to nearest enemy podium.
--   • success  → attempt steal prompts, then immediately queue return home.
--   • failure  → isDueling=false, returningHome=false → next tick retries enemy podium.
-- Phase B (returningHome=true): walk back to own cash pad.
--   • success/failure → reset flags, next tick starts Phase A again.
RunService.Heartbeat:Connect(function()
    if not AutoDuelEnabled then return end

    -- Always keep health up
    local char = LocalPlayer.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum and hum.Health <= 0 then hum.Health = hum.MaxHealth end
    end

    if isDueling then return end

    if returningHome then
        -- Phase B: return to own cash pad
        local cashPadPos = getMyPlotCashPad()
        if not cashPadPos then
            -- Can't find cash pad; reset so we retry enemy podium
            returningHome = false
            return
        end
        isDueling = true
        task.spawn(function()
            tweenToPosition(cashPadPos, RETURN_SPEED, function(success)
                stopAllPathfinding()
                isDueling = false
                returningHome = false   -- Either way, go back to Phase A
                currentDuelTarget = nil
            end)
        end)
    else
        -- Phase A: go to nearest enemy podium
        local nearestPlot = getNearestEnemyPlot()
        if not nearestPlot then return end
        isDueling = true
        currentDuelTarget = nearestPlot.plot.Name
        task.spawn(function()
            local targetPos = nearestPlot.podium:GetPivot().Position
            tweenToPosition(targetPos, PODIUM_10_WALK_SPEED, function(success)
                stopAllPathfinding()
                isDueling = false
                currentDuelTarget = nil
                if success then
                    -- Attempt to fire any nearby steal prompts
                    local grabPromptsList = _G._222Hub_grabPrompts
                    if grabPromptsList then
                        for _, p in ipairs(grabPromptsList) do
                            if p and p.Parent and (p.ActionText or ""):lower():find("steal") then
                                local obj = p.Parent
                                while obj do
                                    if obj:IsA("Model") then
                                        if (obj:GetPivot().Position - targetPos).Magnitude <= 15 then
                                            p.HoldDuration = 0
                                            p.RequiresLineOfSight = false
                                            pcall(function() fireproximityprompt(p) end)
                                        end
                                        break
                                    elseif obj:IsA("BasePart") then
                                        if (obj.Position - targetPos).Magnitude <= 15 then
                                            p.HoldDuration = 0
                                            p.RequiresLineOfSight = false
                                            pcall(function() fireproximityprompt(p) end)
                                        end
                                        break
                                    end
                                    obj = obj.Parent
                                end
                            end

local function getInternalTable()
    local Packages = ReplicatedStorage:FindFirstChild("Packages")
    if not Packages then return nil end
    local SynchronizerModule = Packages:FindFirstChild("Synchronizer")
    if not SynchronizerModule then return nil end
    local success, synchronizer = pcall(require, SynchronizerModule)
    if not success or not synchronizer then return nil end
    local GetMethod = synchronizer.Get
    if type(GetMethod) ~= "function" then return nil end
    for i = 1, 5 do
        local success, upvalue = pcall(getupvalue, GetMethod, i)
        if success and type(upvalue) == "table" then
            if upvalue.___private or upvalue.___channels or upvalue.___data then return upvalue end
            for k, v in pairs(upvalue) do
                if type(k) == "string" and k:match("^Plot_") or type(v) == "table" then return upvalue end
            end
        end
    end
    local success, env = pcall(getfenv, GetMethod)
    if success and env and env.self then return env.self end
    return nil
end

local function stealthGet(plotName)
    if not plotName or type(plotName) ~= "string" then return nil end
    if SynchronizerInternal._cache[plotName] == false then return nil end
    if SynchronizerInternal._dataTable then
        local keys = {plotName, "Plot_" .. plotName, "Plot" .. plotName, plotName .. "_Channel", "Channel_" .. plotName}
        for _, key in ipairs(keys) do
            if SynchronizerInternal._dataTable[key] then
                SynchronizerInternal._cache[plotName] = SynchronizerInternal._dataTable[key]
                return SynchronizerInternal._dataTable[key]
            end
        end
        for k, v in pairs(SynchronizerInternal._dataTable) do
            if type(k) == "string" and (k == plotName or k:find(plotName, 1, true)) and type(v) == "table" then
                SynchronizerInternal._cache[plotName] = v
                return v
            end
        end
    end
    SynchronizerInternal._cache[plotName] = false
    return nil
end

local function stealthGetFromChannel(channel, key)
    if not channel then return nil end

        elseif channel.get and type(channel.get) == "function" then return channel:get(key)
        elseif rawget(channel, key) then return rawget(channel, key) end

    return nil
end

local function refreshPlotData(plot)
    local plotName = plot.Name
    local channel = stealthGet(plotName)
    if not channel then
        for uid, animal in pairs(animals) do
            if animal.plot == plotName then animals[uid] = nil end
        end
        return {}
    end
    local list = stealthGetFromChannel(channel, "AnimalList")
    if not list then
        for uid, animal in pairs(animals) do
            if animal.plot == plotName then animals[uid] = nil end
        end
        return {}
    end
    local currentAnimals = {}
    local plotAnimals = {}
    local isMyPlot = false
    local plots = Workspace:FindFirstChild("Plots")
    if plots and plots:FindFirstChild(plotName) then
        local sign = plots[plotName]:FindFirstChild("PlotSign")
        if sign then
            local yourBase = sign:FindFirstChild("YourBase")
            if yourBase and yourBase:IsA("BillboardGui") then
                isMyPlot = yourBase.Enabled == true
            end
        end
    end
    if isMyPlot then return {} end
    for slot, data in pairs(list) do
        if type(data) ~= "table" then continue end
        local uid = plotName .. "_" .. slot
        currentAnimals[uid] = true
        local success3, gen = pcall(function()
            return AnimalsShared:GetGeneration(data.Index, data.Mutation, data.Traits, nil)
        end)
        if success3 and gen then
            local info = AnimalsData[data.Index] or {DisplayName = data.Index}
            plotAnimals[uid] = {uid = uid, name = info.DisplayName or data.Index, gen = gen, plot = plotName, slot = tostring(slot), isNotMyPlot = true}
        end
    end
    for uid, animal in pairs(animals) do
        if animal.plot == plotName and not currentAnimals[uid] then animals[uid] = nil end
    end
    return plotAnimals
end

local function recompute()
    local best = findBestAnimal()
    if best and (not currentBest or best.uid ~= currentBest.uid) then
        createESP(best)
    elseif best and currentBest and best.uid == currentBest.uid and best.gen ~= currentBest.gen then
        currentBest.gen = best.gen
        updateESPValue()
    elseif not best and currentBest then
        clearESP()
    elseif best and not currentBest then
        createESP(best)
    end
end

local function findNearestPlayer()
    local char = LocalPlayer.Character
    if not char then return nil end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    local nearest = nil
    local minDist = math.huge
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer then
            local targetChar = plr.Character
            local targetHRP = targetChar and targetChar:FindFirstChild("HumanoidRootPart")
            if targetHRP then
                local dist = (hrp.Position - targetHRP.Position).Magnitude
                if dist < minDist then minDist = dist; nearest = plr end
            end
        end
    end
    return nearest
end

_G.FollowTargetDirection = nil
_G.FollowSpeed = nil

local function startFollowing()
    if FollowConnection then FollowConnection:Disconnect() end
    FollowConnection = RunService.Heartbeat:Connect(function()
        if not FollowEnabled then
            _G.FollowTargetDirection = nil
            _G.FollowSpeed = nil
            return
        end
        local char = LocalPlayer.Character
        if not char then return end
        local hrp = char:FindFirstChild("HumanoidRootPart")
        local hum = char:FindFirstChildOfClass("Humanoid")
        if not hrp or not hum then return end
        if not FollowTarget or not FollowTarget.Parent then FollowTarget = findNearestPlayer() end
        if not FollowTarget then
            _G.FollowTargetDirection = nil
            _G.FollowSpeed = nil
            return
        end
        local targetChar = FollowTarget.Character
        if not targetChar then
            FollowTarget = nil
            _G.FollowTargetDirection = nil
            _G.FollowSpeed = nil
            return
        end
        local targetHRP = targetChar:FindFirstChild("HumanoidRootPart")
        if not targetHRP then
            FollowTarget = nil
            _G.FollowTargetDirection = nil
            _G.FollowSpeed = nil
            return
        end
        local direction = targetHRP.Position - hrp.Position
        local distance = direction.Magnitude
        local speedMultiplier = 1
        if distance < SLOW_DISTANCE then speedMultiplier = math.max(0.6, distance / SLOW_DISTANCE) end
        local currentSpeed = FOLLOW_SPEED * speedMultiplier
        if distance > STOP_DISTANCE and direction.Magnitude > 0 then
            _G.FollowTargetDirection = direction.Unit
            _G.FollowSpeed = currentSpeed
        else
            _G.FollowTargetDirection = nil
            _G.FollowSpeed = nil
        end
        if distance <= STOP_DISTANCE then
            hrp.AssemblyLinearVelocity = targetHRP.AssemblyLinearVelocity
            return
        end
        if direction.Magnitude > 0 then
            direction = direction.Unit
            local currentVel = hrp.AssemblyLinearVelocity
            local targetVel = direction * currentSpeed
            local smoothVel = currentVel:Lerp(targetVel, 0.4)
            hrp.AssemblyLinearVelocity = smoothVel
        end
    end)
end

local function dropAnimalLogic()
    local char = LocalPlayer.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum then return end

    local originalCF = hrp.CFrame

    -- Suppress both Anti-Hit and Anti-Ragdoll for the duration
    ah_suppressForDrop = true
    local wasAntiRagdoll = MobileAntiRagdollEnabled
    MobileAntiRagdollEnabled = false

    -- Re-enable blocked states so our state changes actually replicate
    for state in pairs(BlockedStates) do
        pcall(function() hum:SetStateEnabled(state, true) end)
    end

    task.spawn(function()
        -- Step 1: Physics state + upward launch
        hum.PlatformStand = true
        pcall(function() hum:ChangeState(Enum.HumanoidStateType.Physics) end)
        hrp.AssemblyLinearVelocity = Vector3.new(0, 120, 0)
        task.wait(0.1)

        -- Step 2: FallingDown — same state a bat hit produces
        pcall(function() hum:ChangeState(Enum.HumanoidStateType.FallingDown) end)
        task.wait(0.15)

        -- Step 3: Disable Motor6Ds to fully replicate ragdoll
        for _, v in ipairs(char:GetDescendants()) do
            if v:IsA("Motor6D") then v.Enabled = false end
        end
        task.wait(0.1)

        -- Step 4: Re-enable motors and recover
        for _, v in ipairs(char:GetDescendants()) do
            if v:IsA("Motor6D") then v.Enabled = true end
        end
        hum.PlatformStand = false
        pcall(function() hum:ChangeState(Enum.HumanoidStateType.GettingUp) end)
        task.wait(0.05)
        pcall(function() hum:ChangeState(Enum.HumanoidStateType.Running) end)

        -- Step 5: Snap back
        task.wait(0.1)
        if hrp and hrp.Parent then
            hrp.CFrame = originalCF
            hrp.AssemblyLinearVelocity = Vector3.zero
        end

        -- Step 6: Fallback rapid toggle if still holding
        task.wait(0.2)
        local stillHolding = false
        for _, obj in ipairs(workspace:GetChildren()) do
            if obj:IsA("Model") and obj ~= char then
                for _, v in ipairs(obj:GetDescendants()) do
                    if v:IsA("WeldConstraint") or v:IsA("Weld") then
                        local p0, p1
                        pcall(function() p0 = v.Part0 end)
                        pcall(function() p1 = v.Part1 end)
                        if p0 == hrp or p1 == hrp then stillHolding = true; break end
                    end
                end
            end
            if stillHolding then break end
        end

        if stillHolding then
            for i = 1, 5 do
                hum.PlatformStand = true
                task.wait(0.05)
                hum.PlatformStand = false
                task.wait(0.05)
            end
            if hrp and hrp.Parent then
                hrp.CFrame = originalCF
                hrp.AssemblyLinearVelocity = Vector3.zero
            end
        end

        -- Restore everything
        task.wait(0.3)
        MobileAntiRagdollEnabled = wasAntiRagdoll
        UpdateButtonState(AntiRagdollBtn, MobileAntiRagdollEnabled, "Anti Ragdoll [ON]", "Anti Ragdoll [OFF]")
        if wasAntiRagdoll then
            for state in pairs(BlockedStates) do
                pcall(function() hum:SetStateEnabled(state, false) end)
            end
        end
        ah_suppressForDrop = false
    end)
end

local function removeAnimations(character)
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local animator = humanoid and humanoid:FindFirstChild("Animator")
    if animator then
        for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
            track:Stop(0)
        end
    end
    local animate = character:FindFirstChild("Animate")
    if animate then animate:Destroy() end
end

    task.wait(0.5)
    removeAnimations(char)
end)

    removeAnimations(LocalPlayer.Character)
end

-- ================= REMOVE PLAYER TEXTURES =================
local function removeTextures(character)
    for _, obj in ipairs(character:GetDescendants()) do
        if obj:IsA("Shirt") or obj:IsA("Pants") or obj:IsA("ShirtGraphic") then
            obj:Destroy()
        elseif obj:IsA("SpecialMesh") then
            obj.TextureId = ""
        elseif obj:IsA("Decal") or obj:IsA("Texture") then
            obj:Destroy()
        elseif obj:IsA("MeshPart") then
            obj.TextureID = ""
        end
    end
end

    if plr.Character then removeTextures(plr.Character) end
    plr.CharacterAdded:Connect(function(char)
        task.wait(1)
        removeTextures(char)
    end)
end
