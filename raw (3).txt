--// Keek Duel FULL GUI

local plr = game.Players.LocalPlayer
local rs = game:GetService("RunService")
local uis = game:GetService("UserInputService")
local cg = game:GetService("CoreGui")

-- STATES
local Enabled = {
    Left = false,
    ["Infinite Jump"] = false,
    Speed = false,
    SpinBot = false,
    AntiRagdoll = false,
    Hitbox = false,
    ESP = false
}

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "Keek Duel"
gui.Parent = cg

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 260, 0, 350)
frame.Position = UDim2.new(0.5, -130, 0.5, -175)
frame.BackgroundColor3 = Color3.fromRGB(20,20,20)
frame.Parent = gui

-- BUTTON CREATOR
local function button(name, y)
    local b = Instance.new("TextButton")
    b.Size = UDim2.new(0.9,0,0,30)
    b.Position = UDim2.new(0.05,0,0,y)
    b.Text = name..": OFF"
    b.BackgroundColor3 = Color3.fromRGB(40,40,40)
    b.TextColor3 = Color3.new(1,1,1)
    b.Parent = frame

    b.MouseButton1Click:Connect(function()
        Enabled[name] = not Enabled[name]
        b.Text = name..": "..(Enabled[name] and "ON" or "OFF")
    end)
end

-- BUTTONS
button("Left", 10)
button("Infinite Jump", 50)
button("Speed", 90)
button("SpinBot", 130)
button("AntiRagdoll", 170)
button("Hitbox", 210)
button("ESP", 250)

-- RED SPARK
local function spark(char)
    local root = char:WaitForChild("HumanoidRootPart")
    local p = Instance.new("ParticleEmitter")
    p.Color = ColorSequence.new(Color3.fromRGB(255,0,0))
    p.Rate = 20
    p.Parent = root
end

if plr.Character then spark(plr.Character) end
plr.CharacterAdded:Connect(spark)

-- INFINITE JUMP
uis.JumpRequest:Connect(function()
    if Enabled["Infinite Jump"] then
        local hum = plr.Character and plr.Character:FindFirstChildOfClass("Humanoid")
        if hum then
            hum:ChangeState(Enum.HumanoidStateType.Jumping)
        end
    end
end)

-- LOOP FEATURES
rs.RenderStepped:Connect(function()
    local char = plr.Character
    if not char then return end

    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")

    if not hrp or not hum then return end

    -- LEFT (was Life)
    if Enabled.Left then
        hrp.CFrame = hrp.CFrame * CFrame.new(-0.2,0,0)
    end

    -- SPEED
    if Enabled.Speed then
        hum.WalkSpeed = 32
    else
        hum.WalkSpeed = 16
    end

    -- SPINBOT
    if Enabled.SpinBot then
        hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.rad(25), 0)
    end

    -- ANTI RAGDOLL
    if Enabled.AntiRagdoll then
        hum:ChangeState(Enum.HumanoidStateType.GettingUp)
    end

    -- HITBOX EXPANDER
    if Enabled.Hitbox then
        for _, v in pairs(game.Players:GetPlayers()) do
            if v ~= plr and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
                v.Character.HumanoidRootPart.Size = Vector3.new(5,5,5)
                v.Character.HumanoidRootPart.Transparency = 0.5
            end
        end
    end

    -- ESP
    if Enabled.ESP then
        for _, v in pairs(game.Players:GetPlayers()) do
            if v ~= plr and v.Character and v.Character:FindFirstChild("Head") then
                if not v.Character.Head:FindFirstChild("ESP") then
                    local bill = Instance.new("BillboardGui", v.Character.Head)
                    bill.Name = "ESP"
                    bill.Size = UDim2.new(0,100,0,40)
                    bill.AlwaysOnTop = true

                    local txt = Instance.new("TextLabel", bill)
                    txt.Size = UDim2.new(1,0,1,0)
                    txt.Text = v.Name
                    txt.TextColor3 = Color3.new(1,0,0)
                    txt.BackgroundTransparency = 1
                end
            end
        end
    end
end)
