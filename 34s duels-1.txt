    local function canFastMove(hrp, hum, dir)
        if hrp.Anchored then return false end
        if hum.WalkSpeed < 5 then return false end
        local params = RaycastParams.new()
        params.FilterDescendantsInstances = {LocalPlayer.Character}
        params.FilterType = Enum.RaycastFilterType.Exclude
        local flatDir = Vector3.new(dir.X, 0, dir.Z).Unit
        local result = workspace:Raycast(hrp.Position, flatDir * 4, params)
        if result and result.Instance and result.Instance.CanCollide then
            return false
        end
        return true
    end

    local optConnections = {}

    local function applyOpt(obj)
        if not obj:IsDescendantOf(Workspace) and not obj:IsDescendantOf(Lighting) then return end
        if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Sparkles") or
           obj:IsA("Smoke") or obj:IsA("Fire") or obj:IsA("BloomEffect") or obj:IsA("BlurEffect") or
           obj:IsA("SunRaysEffect") or obj:IsA("ColorCorrectionEffect") or obj:IsA("DepthOfFieldEffect") or
           obj:IsA("Accessory") or obj:IsA("PointLight") or obj:IsA("SpotLight") or obj:IsA("SurfaceLight") then
            if not optDestroyedCache[obj] then
                optDestroyedCache[obj] = true
                pcall(function() obj:Destroy() end)
            end
            return
        end
        if obj:IsA("Decal") or obj:IsA("Texture") then 
            pcall(function() obj:Destroy() end) 
            return 
        end
        if obj:IsA("BasePart") or obj:IsA("MeshPart") then
            pcall(function()
                obj.Material = Enum.Material.SmoothPlastic
                obj.Reflectance = 0
                obj.CastShadow = false
            end)
        end
    end

    local function getPos(prompt)
        local p = prompt.Parent
        if p:IsA("BasePart") then return p.Position end
        if p:IsA("Model") then
            local prim = p.PrimaryPart or p:FindFirstChildWhichIsA("BasePart")
            return prim and prim.Position
        end
        if p:IsA("Attachment") then return p.WorldPosition end
        local part = p:FindFirstChildWhichIsA("BasePart", true)
        return part and part.Position
    end
    
    local function findNearestStealPrompt()
        local hrp = getHRP()
        if not hrp then return nil end
        local plots = Workspace:FindFirstChild("Plots")
        if not plots then return nil end
        local myPos = hrp.Position
        local nearest = nil
        local nearestDist = math.huge
        for _, plot in ipairs(plots:GetChildren()) do
            for _, obj in ipairs(plot:GetDescendants()) do
                if obj:IsA("ProximityPrompt") and obj.Enabled and obj.ActionText == "Steal" then
                    local pos = getPos(obj)
                    if pos then
                        local dist = (myPos - pos).Magnitude
                        if dist <= CONFIG.STEAL_RADIUS and dist <= obj.MaxActivationDistance and dist < nearestDist then
                            nearest = obj
                            nearestDist = dist
                        end
                    end
                end
            end
        end
        return nearest
    end

    local DisabledRemotes = {}
    local RemoteWatcher = nil
    local RemoteKeywords = { "useitem", "combatservice", "ragdoll" }
    local function KillRemote(remote)
        if not getconnections or not remote:IsA("RemoteEvent") then return end
        if DisabledRemotes[remote] then return end
        local name = remote.Name:lower()
        for _, key in ipairs(RemoteKeywords) do
            if name:find(key) then
                DisabledRemotes[remote] = {}
                for _, c in ipairs(getconnections(remote.OnClientEvent)) do
                    if c.Disable then c:Disable() table.insert(DisabledRemotes[remote], c) end
                end
                break
            end
        end
    end

        pcall(function()
            Controls:Enable()

        end)
    end
    local function fastSafeDashToSpot(targetPos)
        if not rootPart or not humanoid then return end
        isDashing = true
        for _, obj in pairs(rootPart:GetChildren()) do
            if obj:IsA("BodyVelocity") or obj:IsA("LinearVelocity") or obj:IsA("VectorForce") then obj:Destroy() end
        end
        rootPart.Anchored = true
        humanoid.PlatformStand = true
        local startCFrame = rootPart.CFrame
        local targetCFrame = CFrame.new(targetPos + Vector3.new(0, 1, 0))
        for i = 1, dashFrames do
            if not rootPart then break end
            local alpha = i / dashFrames
            rootPart.CFrame = startCFrame:Lerp(targetCFrame, alpha)
            rootPart.AssemblyLinearVelocity = Vector3.zero
            rootPart.AssemblyAngularVelocity = Vector3.zero
            RunService.RenderStepped:Wait()
        end
        if rootPart then
            rootPart.CFrame = targetCFrame
            task.wait(0.02)
            rootPart.Anchored = false
        end
        humanoid.PlatformStand = false
        humanoid:ChangeState(Enum.HumanoidStateType.Running)
        if rootPart then rootPart.AssemblyLinearVelocity = Vector3.zero end
        isDashing = false
    end
    local function onHitDetected()
        if isProcessingHit or not (CONFIG.AUTO_SENT_TP_ON_HIT or CONFIG.AUTO_ACCEPTED_TP_ON_HIT) then return end
        isProcessingHit = true
        refreshChar()
        local s1, s2
        if CONFIG.AUTO_SENT_TP_ON_HIT then s1 = SENT_S1 s2 = SENT_S2
        elseif CONFIG.AUTO_ACCEPTED_TP_ON_HIT then s1 = ACC_S1 s2 = ACC_S2 end
        if s1 and s2 then
            fastSafeDashToSpot(s1)
            task.wait(0.4)
            if rootPart and humanoid and humanoid.Health > 0 then fastSafeDashToSpot(s2) end
        end
        task.wait(0.3)
        isProcessingHit = false
    end

    local function antiIsRagdoll()
        if not antiCharCache.hum then return false end
        local s = antiCharCache.hum:GetState()
        if s == Enum.HumanoidStateType.Physics or s == Enum.HumanoidStateType.Ragdoll or s == Enum.HumanoidStateType.FallingDown then return true end
        local et = LocalPlayer:GetAttribute("RagdollEndTime")
        if et then
            local n = workspace:GetServerTimeNow()
            if (et - n) > 0 then return true end
        end
        return false
    end

                pcall(function() d:Destroy() end)
            end
        end
    end

    RunService.Stepped:Connect(function()
        if isDashing then return end
        if CONFIG.UP_DOWN_FLOAT or CONFIG.FLOAT_FOLLOW or CONFIG.FLOAT then return end
        local hrp = getHRP()

            if wp then
                local flatTarget = Vector3.new(wp.position.X, hrp.Position.Y, wp.position.Z)
                local dir = flatTarget - hrp.Position
                local dist = dir.Magnitude
                if dist < 5 then
                    if duelPathIndex == 4 then
                        grabSlowdownEndTime = tick() + SLOWDOWN_DURATION
                    end
                    if duelPathIndex == #duelPath then
                        duelPathIndex = 1
                        return
                    end
                    duelPathIndex = duelPathIndex + 1
                else
                    local flatDir = dir.Unit
                    hum:Move(flatDir)
                    local currentSpeed = wp.speed
                    if tick() < grabSlowdownEndTime then
                        currentSpeed = currentSpeed * SLOWDOWN_MULTIPLIER
                    end
                    if canFastMove(hrp, hum, flatDir) then
                        hrp.AssemblyLinearVelocity = Vector3.new(flatDir.X * currentSpeed, hrp.AssemblyLinearVelocity.Y, flatDir.Z * currentSpeed)
                    end
                end
            else
                duelIsMoving = false
            end
            return
        end
        if patrolMode ~= "none" then
            local waypoints = (patrolMode == "right") and rightWaypoints or leftWaypoints

            local currentPos = hrp.Position

            local currentXZ = Vector3.new(currentPos.X, 0, currentPos.Z)
            local distanceXZ = (targetXZ - currentXZ).Magnitude
            if distanceXZ > 3 then
                local moveDirection = (targetXZ - currentXZ).Unit
                local currentSpeed = (currentWaypoint >= 3) and VELOCITY_SPEED_BACK or VELOCITY_SPEED_OUT
                if tick() < grabSlowdownEndTime then
                    currentSpeed = currentSpeed * SLOWDOWN_MULTIPLIER
                end
                hum:Move(moveDirection)
                if canFastMove(hrp, hum, moveDirection) then
                    hrp.AssemblyLinearVelocity = Vector3.new(
                        moveDirection.X * currentSpeed,
                        hrp.AssemblyLinearVelocity.Y,
                        moveDirection.Z * currentSpeed
                    )
                end
            else
                if currentWaypoint == 2 then
                    grabSlowdownEndTime = tick() + SLOWDOWN_DURATION
                end
                if currentWaypoint == #waypoints then
                    stopMovement()
                    if CONFIG.AUTO_RIGHT and AutoRightObj then AutoRightObj.Set(false) end
                    if CONFIG.AUTO_LEFT and AutoLeftObj then AutoLeftObj.Set(false) end
                else
                    currentWaypoint = currentWaypoint + 1
                end
            end
            return
        end

                    hrp.AssemblyLinearVelocity = Vector3.new(md.X * targetSpeed, hrp.AssemblyLinearVelocity.Y, md.Z * targetSpeed)
                end
            end
        end

        if flAntiKbConn then flAntiKbConn:Disconnect() end

            local now = tick()
            if now - flLastKbCheck < FL_KB_INTERVAL then return end
            flLastKbCheck = now
            local curVel = hrp.AssemblyLinearVelocity
            local curPos = hrp.Position
            local posDelt = (curPos - flLastPos).Magnitude
            flLastPos = curPos
            local horizNow = Vector3.new(curVel.X, 0, curVel.Z).Magnitude
            local horizLast = Vector3.new(flLastSafeVel.X, 0, flLastSafeVel.Z).Magnitude
            local isKb = false
            if horizNow > FL_KB_THRESHOLD and horizNow > horizLast * 4 then isKb = true end
            if math.abs(curVel.Y) > 150 then isKb = true end
            local state = hum:GetState()
            if state == Enum.HumanoidStateType.Ragdoll or state == Enum.HumanoidStateType.FallingDown then isKb = true end
            if posDelt > 10 and horizNow > 50 then isKb = true end
            if isKb then
                if state == Enum.HumanoidStateType.Ragdoll or state == Enum.HumanoidStateType.FallingDown then pcall(function() hum:ChangeState(Enum.HumanoidStateType.GettingUp) end) end
                local char = LocalPlayer.Character
                if char then
                    for _, part in ipairs(char:GetDescendants()) do
                        if part:IsA("BasePart") then
                            pcall(function() part.Velocity = Vector3.zero part.RotVelocity = Vector3.zero end)
                            for _, force in ipairs(part:GetChildren()) do
                                if force:IsA("BodyVelocity") or force:IsA("BodyForce") or force:IsA("BodyAngularVelocity") or force:IsA("BodyGyro") then force:Destroy() end
                            end
                        end
                    end
                end
                hum.PlatformStand = false
                hum.AutoRotate = true
                flLastSafeVel = Vector3.zero
            else
                if state ~= Enum.HumanoidStateType.Freefall and state ~= Enum.HumanoidStateType.FallingDown and state ~= Enum.HumanoidStateType.Ragdoll and horizNow < FL_KB_THRESHOLD then
                    flLastSafeVel = curVel
                end
            end
        end)

                end
            end
            
            PinBtn.MouseButton1Click:Connect(function() UpdatePin(not isPinned) end)
            if opts.Pinned then UpdatePin(true) end
          
            local obj = { 
                SetPinned = UpdatePin,
                GetPinned = function() return isPinned end,
                Container = Container 
            }
            table.insert(AllUIObjects, obj)

        end

        function Window:BalanceUI()
            local count = #AllUIObjects
            local mid = math.floor(count / 2)
            local isOdd = (count % 2 ~= 0)
            for i, obj in ipairs(AllUIObjects) do
                obj.Container.Size = UDim2.new(1, 0, 0, obj.Container.Size.Y.Offset)
                if i <= mid then
                    obj.Container.Parent = LeftCol
                elseif isOdd and i == count then
                    obj.Container.Parent = CenterCol
                    obj.Container.Size = UDim2.new(0.6, 0, 0, obj.Container.Size.Y.Offset)
                else
                    obj.Container.Parent = RightCol
                end
            end
        end
        return Window

    local function refreshRecentConfigs()
        pcall(function()
            if not isfolder or not listfiles then return end
            if not isfolder("NemesisConfigs") then return end
            local files = listfiles("NemesisConfigs")
            local diskConfigs = {}
            for _, file in ipairs(files) do
                local name = file:match("([^/%\\]+)%.json$")
                if name then diskConfigs[name] = true end
            end
            local newRecents = {}
            for _, name in ipairs(recentConfigsMemory) do
                if diskConfigs[name] then
                    table.insert(newRecents, name)
                    diskConfigs[name] = nil
                end
            end
            for name, _ in pairs(diskConfigs) do
                table.insert(newRecents, name)
            end
            recentConfigsMemory = newRecents
            RebuildDropdown()
        end)
    end
