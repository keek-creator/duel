local function drag(f)
    local on,s1,s2=false,nil,nil
    f.InputBegan:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.Touch or i.UserInputType==Enum.UserInputType.MouseButton1 then
            on=true s1=i.Position s2=f.Position
        end
    end)
    f.InputEnded:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.Touch or i.UserInputType==Enum.UserInputType.MouseButton1 then on=false end
    end)
    UIS.InputChanged:Connect(function(i)
        if on and (i.UserInputType==Enum.UserInputType.Touch or i.UserInputType==Enum.UserInputType.MouseMovement) then
            local d=i.Position-s1
            f.Position=UDim2.new(s2.X.Scale,s2.X.Offset+d.X,s2.Y.Scale,s2.Y.Offset+d.Y)
        end
    end)
end

-- ESP
local bills={}
local function clearESP()
    for i=1,#bills do if bills[i] and bills[i].Parent then bills[i]:Destroy() end end
    bills={}
end
local function buildESP()
    clearESP()
    if not ST.esp then return end
    for _,p in pairs(Players:GetPlayers()) do
        if p~=lp and p.Character then
            local hr=p.Character:FindFirstChild("HumanoidRootPart")
            local hm=p.Character:FindFirstChildOfClass("Humanoid")
            if hr and hm then
                local bb=Instance.new("BillboardGui",hr)
                bb.Size=UDim2.new(0,110,0,44) bb.StudsOffset=Vector3.new(0,4,0)
                bb.AlwaysOnTop=true bb.Adornee=hr
                local n=Instance.new("TextLabel",bb)
                n.Size=UDim2.new(1,0,0.5,0) n.BackgroundTransparency=1
                n.Text=p.Name n.TextSize=12 n.Font=Enum.Font.GothamBold
                n.TextColor3=PLT n.TextStrokeTransparency=0 n.TextXAlignment=Enum.TextXAlignment.Center
                local s=Instance.new("TextLabel",bb)
                s.Size=UDim2.new(1,0,0.5,0) s.Position=UDim2.new(0,0,0.5,0)
                s.BackgroundTransparency=1 s.Text="SPD:"..math.floor(hm.WalkSpeed)
                s.TextSize=11 s.Font=Enum.Font.Gotham s.TextXAlignment=Enum.TextXAlignment.Center
                s.TextColor3=Color3.fromRGB(255,230,60) s.TextStrokeTransparency=0
                bills[#bills+1]=bb
            end
        end
    end
end

-- AIMBOT
local function nearest()
    local cam=game:GetService("Workspace").CurrentCamera
    local best,bd=nil,math.huge
    for _,p in pairs(Players:GetPlayers()) do
        if p~=lp and p.Character then
            local hr=p.Character:FindFirstChild("HumanoidRootPart")
            local hm=p.Character:FindFirstChildOfClass("Humanoid")
            if hr and hm and hm.Health>0 then
                local sp,vis=cam:WorldToScreenPoint(hr.Position)
                if vis then
                    local cv=cam.ViewportSize
                    local d=(Vector2.new(sp.X,sp.Y)-Vector2.new(cv.X/2,cv.Y/2)).Magnitude
                    if d<bd then bd=d best=p end
                end
            end
        end
    end
    return best
end

-- FLOAT
local function applyFloat()
    local hr=HRP() local hm=Hum()
    if not hr or not hm then return end
    pcall(function() hm:ChangeState(Enum.HumanoidStateType.Physics) end)
    local bp=hr:FindFirstChild("MECO_BP")
    if not bp then
        bp=Instance.new("BodyPosition",hr)
        bp.Name="MECO_BP" bp.MaxForce=Vector3.new(0,math.huge,0) bp.D=500 bp.P=5000
    end
    bp.Position=hr.Position
end

-- STEAL
local function doSteal()
    local hr=HRP() if not hr then return end
    for _,v in pairs(game:GetService("Workspace"):GetDescendants()) do
        if v:IsA("ProximityPrompt") then
            local n=(v.ActionText..v.ObjectText):lower()
            if n:find("steal") or n:find("grab") or n:find("collect") then
                local pt=v.Parent
                if pt and pt:IsA("BasePart") and (hr.Position-pt.Position).Magnitude<20 then
                    pcall(function() fireproximityprompt(v) end)
                end
            end
        end
    end
end

-- MAIN LOOP
local et=0
RunService.Heartbeat:Connect(function(dt)
    local hm=Hum() local hr=HRP()
    if hm and not ST.unwalk then hm.WalkSpeed=ST.speed end
    if ST.float then applyFloat() end
    if hr then
        if ST.autoL then hr.CFrame=hr.CFrame*CFrame.new(-0.3,0,0) end
        if ST.autoR then hr.CFrame=hr.CFrame*CFrame.new(0.3,0,0) end
        if ST.aim then
            local t=nearest()
            if t and t.Character then
                local th=t.Character:FindFirstChild("HumanoidRootPart")
                if th then hr.CFrame=CFrame.new(hr.Position,Vector3.new(th.Position.X,hr.Position.Y+ST.aimH,th.Position.Z)) end
            end
        end
    end
    if ST.steal then doSteal() end
    et=et+dt
    if et>=2 then et=0 buildESP() end
end)

lp.CharacterAdded:Connect(function()
    task.wait(1)
    if ST.unwalk then local h=Hum() if h then h.WalkSpeed=0 end end
end)
